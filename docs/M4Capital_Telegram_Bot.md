-- Users Table
CREATE TABLE users (
    telegram_id BIGINT PRIMARY KEY,
    email VARCHAR(255) UNIQUE,
    phone VARCHAR(20),
    full_name VARCHAR(100),
    country VARCHAR(50),
    verification_level INT DEFAULT 1,
    kyc_status VARCHAR(20),
    balance DECIMAL(15,2) DEFAULT 0,
    bonus_balance DECIMAL(15,2) DEFAULT 0,
    account_type VARCHAR(20) DEFAULT 'standard',
    leverage INT DEFAULT 100,
    is_vip BOOLEAN DEFAULT FALSE,
    referral_code VARCHAR(20) UNIQUE,
    referred_by VARCHAR(20),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_login TIMESTAMP
);

-- Trades Table
CREATE TABLE trades (
    id SERIAL PRIMARY KEY,
    user_id BIGINT REFERENCES users(telegram_id),
    symbol VARCHAR(20),
    direction VARCHAR(4), -- BUY/SELL
    entry_price DECIMAL(15,5),
    exit_price DECIMAL(15,5),
    volume DECIMAL(15,2),
    stop_loss DECIMAL(15,5),
    take_profit DECIMAL(15,5),
    status VARCHAR(20), -- open, closed, pending
    profit_loss DECIMAL(15,2),
    commission DECIMAL(15,2),
    opened_at TIMESTAMP,
    closed_at TIMESTAMP
);

-- Transactions Table
CREATE TABLE transactions (
    id SERIAL PRIMARY KEY,
    user_id BIGINT REFERENCES users(telegram_id),
    type VARCHAR(20), -- deposit, withdrawal, bonus
    amount DECIMAL(15,2),
    currency VARCHAR(10),
    method VARCHAR(50),
    status VARCHAR(20), -- pending, completed, failed
    transaction_id VARCHAR(100) UNIQUE,
    created_at TIMESTAMP
);

-- Signals Table
CREATE TABLE signals (
    id SERIAL PRIMARY KEY,
    symbol VARCHAR(20),
    direction VARCHAR(4),
    entry_price DECIMAL(15,5),
    stop_loss DECIMAL(15,5),
    take_profit DECIMAL(15,5),
    confidence INT,
    status VARCHAR(20), -- active, hit, stopped
    created_at TIMESTAMP,
    expires_at TIMESTAMP
);

-- Price Alerts Table
CREATE TABLE price_alerts (
    id SERIAL PRIMARY KEY,
    user_id BIGINT REFERENCES users(telegram_id),
    symbol VARCHAR(20),
    target_price DECIMAL(15,5),
    condition VARCHAR(10), -- above, below
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP
);

-- Referrals Table
CREATE TABLE referrals (
    id SERIAL PRIMARY KEY,
    referrer_id BIGINT REFERENCES users(telegram_id),
    referred_id BIGINT REFERENCES users(telegram_id),
    status VARCHAR(20), -- pending, active, rewarded
    commission_earned DECIMAL(15,2) DEFAULT 0,
    created_at TIMESTAMP
);
